```python
@runtime_checkable
class Backend(Protocol):
    @property
    def name(self) -> str: ...
    @property
    def version(self) -> str: ...
    def chat(self, model, messages, stream=False, options=None) -> BackendResponse | StreamResult: ...
    def list_models(self) -> list[dict]: ...
    def unload_model(self, model) -> bool: ...
    def warmup(self, model, timeout) -> bool: ...
    def detect_context_window(self, model) -> int: ...
    def get_model_size(self, model) -> float | None: ...
    def check_connectivity(self) -> bool: ...

class BackendResponse(BaseModel):
    model: str
    content: str
    done: bool
    prompt_eval_count: int = 0
    eval_count: int = 0
    total_duration: float = 0.0      # seconds
    load_duration: float = 0.0       # seconds
    prompt_eval_duration: float = 0.0 # seconds
    eval_duration: float = 0.0       # seconds
    prompt_cached: bool = False

class BackendError(Exception):
    retryable: bool
    original: Exception | None

class StreamResult:
    chunks: Iterator[str]
    response: BackendResponse  # available after iteration

def create_backend(name="ollama") -> Backend: ...
```
